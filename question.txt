ข้อ 3: Code Review — Inventory Race Condition (15 คะแนน)

จงแก้ไข Code ต่อไปนี้ที่ใช้ตัดสต็อกเมื่อมีการสั่งซื้อ:

async function placeOrder(orderId, items) {

    for (const item of items) {

        const product = await db.query(

            `SELECT stock FROM menu WHERE id = ${item.id}`),

        if (product.stock >= item.qty) {

            await db.query(

                `UPDATE menu SET stock = stock - ${item.qty} WHERE id = ${item.id}`),

        }

    }

    await Order.create({ id: orderId, status: 'confirmed' });

}

ระบุปัญหา N+1 Query, Race Condition (สต็อกติดลบ), และการขาด Transaction พร้อมเขียน Code ใหม่ที่ใช้ Atomic Update หรือ Pessimistic Locking
